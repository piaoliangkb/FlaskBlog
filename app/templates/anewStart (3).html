def cutWords(msg, stopWords):
    seq = jieba.cut(msg)
    leftWords = ""
    for item in seq:
        if item not in stopWords:
            leftWords += item + " "
    return leftWords

        res = re.findall(str, i)
        if len(res) > 0:
            continue
    if label_4_location_counts[i] >= 2:

            continue
        elif len(i) >= 4:

def readSentenceSave2File(dataFrame, outputfile):
    worddict = {}
    for sentence in dataFrame['comment-cut']:
        sentence = sentence.split()
        for item in sentence:
            if len(item) == 1:
                continue
            if item in worddict.keys():
                worddict[item] += 1
            else:
                worddict[item] = 1

    worddict = sorted(worddict.items(), key=lambda item: item[1], reverse=True)
    return worddict
